‚ö†Ô∏è Checks Before Submission
If user selects rows more than the available rows in the current page through custom Row Selection, then you must not pre fetch next (other) page rows and store them in a variable.

This is to prevent prefetching of pages in advance, which can slow down the application and also may lead to out of memory issue.

Instead, you must implement a strategy to handle custom row selection and deselection for rows across different page.

Any submission with this issue is instantly discarded.

const handleCustomRowSelection = async (): Promise<void> => {
    const count = parseInt(selectCount);
    
    if (!count || count <= 0) {
      alert('Please enter a valid number');
      return;
    }

    const newSelectedIds = new Set<number>();
    const newDeselectedIds = new Set<number>();
    let collected = 0;
    let currentPage = 1;

    while (collected < count) {
      try {
        const response = await fetch(
          `https://api.artic.edu/api/v1/artworks?page=${currentPage}`
        );
        const data: ApiResponse = await response.json();
        
        const needed = count - collected;
        const toSelect = data.data.slice(0, needed);
        
        toSelect.forEach(artwork => {
          newSelectedIds.add(artwork.id);
          collected++;
        });
        
        if (toSelect.length < needed && data.data.length < rowsPerPage) {
          break;
        }
        
        currentPage++;
        
      } catch (error) {
        console.error('Error fetching artworks for bulk selection:', error);
        break;
      }
    }

    setSelectedRowIds(newSelectedIds);
    setDeselectedRowIds(newDeselectedIds);
    setShowOverlay(false);
    setSelectCount('');
  };
